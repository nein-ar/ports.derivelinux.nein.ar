#!/bin/sh -ue
NAME=ffmpeg
VERSION=git:n6.1.4
RELEASE=1
SOURCE="https://git.ffmpeg.org/ffmpeg.git"

build() {
	./configure \
		--pkg-config-flags=--static \
		--extra-cflags="-I$PREFIX/include" \
		--extra-ldflags="-L$PREFIX/lib -static" \
		--extra-libs="-lpthread -lm" \
		--extra-ldexeflags=-static \
		--host-ldflags=-static \
		--disable-debug \
		--disable-libxml2 \
		--enable-static \
		--enable-gpl \
		--enable-libass \
		--enable-libdrm \
		--enable-libmp3lame \
		--enable-libopus \
		--enable-libvpx \
		--enable-libwebp \
		--enable-libx264 \
		--enable-libx265 \
		--enable-libxvid \
		--enable-libdav1d \
		--enable-nonfree \
		--enable-version3 \
		--disable-doc \
		--disable-htmlpages \
		--enable-manpages \
		--disable-podpages \
		--disable-txtpages \
		--enable-libtls
	gmake PREFIX="/usr" -j"$NPROC"
	gmake PREFIX="/usr" DESTDIR="$PKG" install
}

. ${0%/*}/../../libsh/libdmake.sh
